# 🧮 Sample DAX Measures : 

# Total Revenue
Total Revenue = SUM(Orders[Revenue])

# Average Order Value
AOV = DIVIDE([Total Revenue], [Total Orders])

# Rush Order Percentage
Rush Orders =
CALCULATE(
    COUNT(Orders[Order ID]),
    Orders[Rush Shipment] = "Yes"
)

Rush % = DIVIDE([Rush Orders], [Total Orders])
